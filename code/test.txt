fun IsPrime(n: Int): Int {
    if (n <= 1) {
        return 0;
    }
    var i = 2;
    while (i * i <= n) {
        var prod = i * (n / i);
        if (prod == n) {
            return 0;
        }
        i = i + 1;
    }
    return 1;
}

fun CountPrimes(limit: Int): Int {
    var count = 0;
    var i = 2;
    while (i <= limit) {
        if (IsPrime(i) == 1) {
            count = count + 1;
        }
        i = i + 1;
    }
    return count;
}

fun main(): Int {
    var n = 100000;
    var primeCount = CountPrimes(n);
    return primeCount;
}
